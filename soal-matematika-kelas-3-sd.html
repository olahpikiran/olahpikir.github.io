[tex]3 \div 0,3 = 10[/tex]

[tex] = 3 \div 0,3[/tex]

[tex] = 3 \div \frac{3}{10} [/tex]

[tex] = 3 \times \frac{10}{3} [/tex]

[tex] = \frac{30}{3} [/tex]

[tex] = 10[/tex]
