  0 = 9 - 6t²
6t² = 9
  t² = [tex] \frac{9}{6} [/tex] = [tex] \frac{3}{2} [/tex]
  t  = [tex] \sqrt{ \frac{3}{2} } [/tex]
  t  = [tex] \frac{1}{2} \sqrt{6} [/tex]
jadi waktu max = 9 .[tex]( \frac{1}{2} \sqrt{6}) [/tex] - 2 .[tex] ( \frac{1}{2} \sqrt{6}) ^{3} [/tex]   
                       = [tex] \frac{9}{2} \sqrt{6} [/tex] - 2 . [tex] \frac{6}{8} \sqrt{6} [/tex]
                       = [tex] \frac{9}{2} \sqrt{6} [/tex] - [tex] \frac{3}{2} \sqrt{6} [/tex]
                       = [tex] \frac{6}{2} \sqrt{6} [/tex]
