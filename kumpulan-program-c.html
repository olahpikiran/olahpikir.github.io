30. Buatlah program dalam bahasa c mencari batasan angka diamond tengahnya bolong menggunakan karakter * seperti contoh dibwah. Tolong dibantu hari ini di kumpul.............. ****** ****** *****   ***** ****     **** ***       *** **         ** *            * **          ** ***        *** ****      **** *****    ***** ******  ******


#include <stdio.h>
int main(){
    int n;
    scanf("%d",&n);
    n+=1;
    for (int i=1; i<=n-1; i++){
        for (int j=1; j<=n*2-1; j++){
            if (n>=j-i+1 && n<=j+i-1){
                printf(" ");
            }
            else printf("*");
        }
        printf("\n");
    }
    for (int i=n-2; i>=1; i--){
        for (int j=1; j<=n*2-1; j++){
            if (n>=j-i+1 && n<=j+i-1){
                printf(" ");
            }
            else printf("*");
        }
        printf("\n");
    }
}
