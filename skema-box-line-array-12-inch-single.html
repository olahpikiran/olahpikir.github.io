using namespace std;

class Setter{

   protected:

   int angkaUrut[10];

   int m1,m2,m3,m4,m5;

};

class Getter : Setter{

   public:

   void Switching(){

       m2 = angkaUrut[1];

       angkaUrut[1] = angkaUrut[9];

       angkaUrut[9] = m2;

       

       m3 = angkaUrut[2];

       angkaUrut[2] = angkaUrut[8];

       angkaUrut[8] = m3;

       

       m4 = angkaUrut[3];

       angkaUrut[3] = angkaUrut[7];

       angkaUrut[7] = m4;

       

       m5 = angkaUrut[4];

       angkaUrut[4] = angkaUrut[6];

       angkaUrut[6] = m5;

       

   }

   

   void Input(){

       cout << "Hasil Input: " << endl << "{";

   for(int i=0;i<=9;i++){

       cout << angkaUrut[i];

           cout << ", ";

       }

   }

   cout << "}" << endl;

   }

   

   void Output(){

   for(int i=1;i<=10;i++){

       cout << angkaUrut[i];

       if(i!=10){

           cout << ", ";

       }

   }

   cout << "}" << endl;

   }

   

};

27. baris program untuk mengetahui panjang array multidimensi? misalnya dalam array nilai [ 1, 3] isinya itu {12, 7, 4, 9} panjang array nya = 4


contoh kode:

#include <iostream>

using namespace std;

int main()

{

int arry[3][2]={{0,1},{2,3},{4,5}};

 

cout<<sizeof(arry)/sizeof(int);

 

 

return  0;

}

untuk konversi ke jumlah dalam array terakhir nya adalah:cout<<sizeof(arry)/sizeof(int);

bisa ditampung dengan variabel.

kenapa sizeof?

pada dasarnya sizeof itu berguna untuk menkonversi berapa byte yg sudah dikeluarkan misalkan:

sizeof(int):4;

sizeof(short int):2;

nah sizeof(arry) berguna untuk melihat berapa saja variabel yg sudah digunakan di kondisi ini setiap variabel berjumlah 4 karena bertipe data integer. oleh karena itu kita harus membagi nya dengan basis dari integer/tipe data yang digunakan di array tersebut (dalam kondisi ini saya menggunakan array bertipe integer)

kesimpulan:

jika dalam 1 angka(bertipe integer) dibutuhkan =4 Byte, dalam kondisi ini array yang saya buat ada 6 berarti yang sudah dialokasikan adalah 6x4 karena bertipe data integer yaitu menjadi 24Byte karena itu kita harus membagi nya dengan integer itu sendiri yaitu bernilai 4 Byte

maka konversi dapat dirumuskan :sizeof(arry)/sizeof(int);

29. 1.buatlah program yang menerima input jumlah array, isi array, key yang ingin dicari. cetak ke layar index posisi elemen dari key yang ingin dicari. contoh hasil program: masukkan jumlah elemen array: 4 array ke 0. : 5 array ke 1. : 12 array ke 2. : 25 array ke 3. : 10 masukkan key yang ingin dicari: 12 key ada di posisi index ke: 1


Materi: Bahasa Pemrograman C++

Penjelasan/Algoritma:

Program dimulai.Input elemenArray, arrayKe, keyDicari.Perulangan untuk input arrayKe.Perulangan indeks arrayKe hingga keyDicari ditemukan.Cetak output.Program selesai.

Kode Program:

//Diprogram oleh M.Fazri Nizar

//Program dimulai

#include <iostream>

#include <string>

using namespace std;

int main() {

   int elemenArray, i, j = 0;

   

   cout << "\n\t\t [?] Diprogram oleh: M.Fazri Nizar [?]\n\n Masukkan jumlah elemen array: ";

   cin >> elemenArray;

   string arrayKe[100], keyDicari;

   

   for(i = 0; i < elemenArray; i++) {

       cout << "\nArray ke-" << i << ": ";

       cin >> arrayKe[i];

   }

   

   cout << "\nMasukkan key yang ingin dicari: ";

   cin >> keyDicari;

   

   while(j <= elemenArray) {

       j++;

       if(arrayKe[j] == keyDicari) {

           cout << "\nKey ada di posisi index ke: " << j << endl;

           break;

       } else if(j == elemenArray) {

           cout << "\nKey tidak ada di posisi index manapun." << endl;

       }

   }

   

   return 0;

}

//Program selesai
