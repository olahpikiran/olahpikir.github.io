Bola besar = x
Bola kecil = y

50x - 30y = 50
50x + 30y = 1550
---------------------------- +
100x = 1600
x = 16

50x - 30y = 50
50(16) - 30y = 50
800 - 30y = 50
800 - 50 = 30y
750 = 30y
y = 25
